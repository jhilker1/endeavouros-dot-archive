** Scripts
*** Toggle Script
This script simply toggles the sidebar.
#+begin_src bash :shebang "#!/usr/bin/env bash\n" :tangle "scripts/toggle-rightbar.sh"
EWW=`which eww`
FILE="$HOME/.cache/eww_launch.sb"
CFG="$HOME/.config/eww/rightbar"
SB_NAME="example"

if [[ ! `pidof eww` ]]; then
    eww daemon
    sleep 1
fi

if [[ ! -f "$FILE" ]]; then
    touch "$FILE"
    $EWW open "$SB_NAME"
else
   $EWW close "$SB_NAME"
   rm -rf "$FILE"
fi
#+end_src
